70void LCD_Init(void)
71{
72 uint8_t tmp = (1 << LCD_BKLight);
73 _delay_ms(15); // oczekiwanie na ustalibizowanie sie napiecia zasilajacego
74
75 PCF8574_WritePort(tmp);
76
77 for(uint8_t i = 0; i < 3; i++) // trzykrotne powtorzenie bloku instrukcji
78 {
79 tmp |= (1 << LCD_E);
80 tmp = (tmp & 0x0F) | (0x03 << LCD_DaTa);
81
82 PCF8574_WritePort(tmp);
83
84 tmp &= ~(1 << LCD_E);
85 PCF8574_WritePort(tmp);
86 _delay_ms(5); // czekaj 5ms
87 }
88
89 tmp |= (1 << LCD_E);
90 tmp = (tmp & 0x0F) | (0x02 << LCD_DaTa);
91 PCF8574_WritePort(tmp);
92
93 tmp &= ~(1 << LCD_E);
94 PCF8574_WritePort(tmp);
95
96 _delay_ms(1); // czekaj 1ms
97 LCD_WriteCommand(HD44780_FUNCTION_SET | HD44780_FONT5x7 | HD44780_TWO_LINE | HD44780_4_BIT); // interfejs 4-bity, 2-linie, znak 5x7
98 LCD_WriteCommand(HD44780_DISPLaY_ONOFF | HD44780_DISPLaY_OFF); // wylaczenie wyswietlacza
99 LCD_WriteCommand(HD44780_CLEaR); // czyszczenie zawartosci pamieci DDRaM
100 _delay_ms(2);
101 LCD_WriteCommand(HD44780_ENTRY_MODE | HD44780_EM_SHIFT_CURSOR | HD44780_EM_INCREMENT);// inkrementaja adresu i przesuwanie kursora
102 LCD_WriteCommand(HD44780_DISPLaY_ONOFF | HD44780_DISPLaY_ON | HD44780_CURSOR_OFF | HD44780_CURSOR_NOBLINK); // wlacz LCD, bez kursora i mrugania
103}
